import getopt, sys

def usage():
	print("Script Usage:")
	print("-h OR --help					: To print Help")
	print("-i <input-file>  OR --input <input-file>	: To provide input file taken from SNOW")
	print("-o <output-file> OR --output <output-file>	: To provide output file name")
	print("-r <report-flag> OR --report <report-flag>	: To provide report flag. report-flag options are 'all','os','db','mw'")
	print("-s <subset-file> OR --subset <subset-file>	: To provide subset sysIDs input file")
